function w = W(k,m,M,x,xstar)
%On commence par calculer P(A)
p = m*x + (1-m)*xstar;
pstar = (1-m)/(M-1)*x + xstar*(m + (M-2)/(M-1)*(1-m));
Pa = PA(k,m,M,p);
Pastar = PA(k,m,M,pstar);
w = k*(1-x)*(m*Pa + (1-m)*Pastar);
end
 
function P = PA(k,m,M,p)
P = 0;
for i = 0:k*M
partialSum = 0;
for j = 0:i
if j > k
P1 = 0;
else
P1 = nchoosek(k, j)*m^(k-j)*(1-m)^j;
end
 
if i-j > k*(M-1)
P2 = 0;
else
P2 = nchoosek(k*(M-1), i-j)*((1-m)/(M-1))^j*(1-(1-m)/(M-1))^(k*(M-1)-j);
end
partialSum = partialSum + P1*P2;
end
P = P + (1 - (1-p)^i)*partialSum;
end
end
 
function mini=minimum_m_W(k,M,x,xstar)
phi=(1+sqrt(5))/2;
a=0;
b=1;
c=b-(b-a)/phi;
d=a+(b-a)/phi;
N_ite=20;
for i=1:1:N_ite
if W(k,c,M,x,xstar)<=W(k,d,M,x,xstar)
b=d;
d=c;
c=b-(b-a)/phi;
else if W(k,c,M,x,xstar)>W(k,d,M,x,xstar)
a=c;
c=d;
d=a+(b-a)/phi;
end
end
mini=(a+b)/2;
end
end

function nouveau_terrain = transition(terrain, M, m)
%Toutes les fonctions renverront un nouvel ecosyst√®me afin de pouvoir
%facilement les comparer entre eux.
nouveau_terrain = terrain;
 
for lieu = 1:M
k = length(terrain{lieu});
%On parcourt les n√©matodes dans l'ordre d√©croissant pour √©viter les
%probl√®mes d'ordonnement
for nem = k:-1:1
%On prend un n√©matode
nematode = nouveau_terrain{lieu}{nem};
%Il va migrer avec la probabilit√© 1-m
if rand() > m
%La migration se fait de fa√ßon uniforme vers l'un des M-1
%lieux
indice_migration = randi([1, M - 1]);
%Si le lieu choisi est celui sur lequel est le n√©matode, on
%prend le lieu suivant
if indice_migration == lieu
indice_migration = indice_migration + 1;
end
%On ajoute le n√©matode au nouveau lieu
nouveau_terrain{indice_migration} = [nouveau_terrain{indice_migration}, {nematode}];
%On le supprime de l'ancien lieu
nouveau_terrain{lieu}{nem} = [];
end
end
end
end

function descendants = fecondation(nematode, k, reproducteur, x, xstar)

%sex_ratio utilise le fait qu'en Matlab, True = 1 et False = 0

%On v√©rifie que le n√©matode est une femelle et qu'il y a des m√¢les sur le
%lieu

%On remplit alors la liste des k descendants en leur donnant la
%m√™me ADN que leur m√®re, le sexe est choisi gr√¢ce au sex_ratio
